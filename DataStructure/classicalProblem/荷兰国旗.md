import java.util.*;

public class Main {
	
	//红白蓝 荷兰国旗问题：
	private final static int red = 0;
	private final static int white = 1;
	private final static int blue = 2;
	
	public static void main(String[] args) {
		int a[] = {0,1,2,0,0,1,2};
		for (int i : a) {
			System.out.print(i+" ");
		}
		sort(a);
		System.out.println();
		for (int i : a) {
			System.out.print(i+" ");
		}
		
	}
	
  /*关键算法
  */
	private static void sort(int a[]) {
		int p = 0;//白球起始分界点
		int q = a.length-1;//白球终止分界点
		int current = p;
		while(current <=q){
				switch (a[current]) {
				case red:
						if(current != p)
							swap(a,current,p);
						current++;
						p++;
					break;
				case white:
						current++;
					break;
				case blue:
					swap(a, current, q);
					q--;
					break;
				default:
					break;
			}
		}	
	}

	private static void swap(int[] a, int current, int p) {
		int temp = a[p];
		a[p] = a[current];
		a[current] = temp;
	}

}
